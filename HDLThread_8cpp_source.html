482 hNodep hw = htmp->child_list.back(); // the wait instruction
483 if (hw->child_list.size() == 0) {
484 hw->set(hNode::hdlopsEnum::hBinop, "=");

487 }

490 // generate waitcounter = wait arg; nextwaitstate = nxtstate; nextstate = waitstate (which is numstates)

492 string waitarg = hnewinstr->getname();
493 // wait counter = wait arg
494 hw->set( hNode::hdlopsEnum::hBinop, "=");
495 hnewinstr->set(hNode::hdlopsEnum::hVarref, nextwaitctr_string);
496 hw->append(new hNode(waitarg, hNode::hdlopsEnum::hLiteral));

537 hNodep hcstmt = new hNode(hNode::hdlopsEnum::hCStmt);
538 hcstmt->append(GenerateBinop(comb_assign, state_string, "0"));
539 hcstmt->append(GenerateBinop(comb_assign, waitnextstate_string, "0"));
540 hcstmt->append(GenerateBinop(comb_assign, waitctr_string, "0"));
