238 std::string generateSensitivityName(
239 const std::vector<SensitivityTupleType> &sense_args) {
240 std::string name{};
241 std::string ef_name{};
242 for (auto const entry : sense_args) {
243 name = name + std::get<0>(entry);
244 auto vd{std::get<3>(entry)};
245 if (vd) { ef_name = vd->getNameInfo().getAsString();}
246 }
247 return ef_name+"__"+name;
248 }

250 public:
255 SenseMapType getSensitivityMap() { return sensitivity_; }

259 void registerMatchers(MatchFinder &finder) {
260 /* clang-format off */
261
262 auto match =
263 cxxConstructorDecl(
264 has(
265 compoundStmt(
266 forEachDescendant(
267 cxxOperatorCallExpr(optionally(
268 hasDescendant(declRefExpr(hasType(cxxRecordDecl(hasName("::sc_core::sc_process_handle")))).bind("process_handle"))
269 )
270 ).bind("opcall")
271 )
272 ).bind("compound_stmt")
273 )
274 ).bind("cxx_constructor_decl");
275
276 /* clang-format on */
278 finder.addMatcher(match, this);
279 }

282 virtual void run(const MatchFinder::MatchResult &result) {
283 // LLVM_DEBUG(llvm::dbgs()
284 DEBUG_WITH_TYPE("matcher-header",
285 llvm::dbgs()
286 << "====== SENSITIVITY MATCHER EXECUTED ======= \n");
287
288 clang::CXXMemberCallExpr *cxx_mcall{};
289 clang::MemberExpr *me_wo_mcall{};
290 clang::ArraySubscriptExpr *array_expr{};
291
293 auto process_handle{const_cast<clang::DeclRefExpr *>(
294 result.Nodes.getNodeAs<clang::DeclRefExpr>("process_handle"))};
295 auto opcall{const_cast<clang::CXXOperatorCallExpr *>(
296 result.Nodes.getNodeAs<clang::CXXOperatorCallExpr>("opcall"))};
297
298 if (opcall) {
299 // Check if there is process handle
300 if (process_handle) {
301 clang::ValueDecl *vd{process_handle->getDecl()};
302 LLVM_DEBUG(llvm::dbgs() << "# DeclRefExpr "
303 << process_handle->getNameInfo().getAsString()
304 << " => " << vd->getNameAsString() << "\n");
305 process_handle_ = process_handle;
306 }
307
308 MatchFinder sense_registry{};
309 SensitiveOperatorCallMatcher sop_matcher{};
310 sop_matcher.registerMatchers(sense_registry);
311 sense_registry.match(*opcall, *result.Context);
312
313 cxx_mcall = sop_matcher.getMemberExprCallExpr();
314 me_wo_mcall = sop_matcher.getMemberExprWithoutCall();
315 array_expr = const_cast<clang::ArraySubscriptExpr *>(
316 sop_matcher.getMemberArraySubscriptExpr());
317 }
318
323
324 if (process_handle_) {
325 if (me_wo_mcall) {
326 MatchFinder call_registry{};
327 CallerCalleeMatcher call_matcher{};
328 call_matcher.registerMatchers(call_registry, process_handle_);
329 call_registry.match(*me_wo_mcall, *result.Context);
330
331 LLVM_DEBUG(call_matcher.dump());
332 auto entry{call_matcher.getCallerCallee()};
333 sensitivity_.insert(
334 SensitivityPairType(generateSensitivityName(entry), entry));
335 }
336
340 if (cxx_mcall) {
341 MatchFinder call_registry{};
342 CallerCalleeMatcher call_matcher{};
343 call_matcher.registerMatchers(call_registry, process_handle_);
344 call_registry.match(*cxx_mcall, *result.Context);
345 LLVM_DEBUG(call_matcher.dump());
346
347 auto entry{call_matcher.getCallerCallee()};
348 sensitivity_.insert(
349 SensitivityPairType(generateSensitivityName(entry), entry));
350 }
351
354 if (array_expr) {
355 llvm::outs() << "@@@@ Parse the array \n";
356 auto me{getArrayMemberExprName(array_expr)};
357 std::string name{me->getMemberDecl()->getNameAsString()};
358 auto entry{std::make_tuple(name, me->getMemberDecl(),
359 const_cast<clang::MemberExpr *>(me),
360 process_handle_, array_expr)};
361 std::vector<
362 std::tuple<std::string, clang::ValueDecl *, clang::MemberExpr *,
363 clang::DeclRefExpr *, clang::ArraySubscriptExpr *>>
364 calls;
365 calls.insert(calls.begin(),
366 std::make_tuple(name, me->getMemberDecl(),
367 const_cast<clang::MemberExpr *>(me),
368 process_handle_, array_expr));
369
370 sensitivity_.insert(
371 SensitivityPairType(generateSensitivityName(calls), calls));
372 }
373 }
374 LLVM_DEBUG(dump());
375 }
