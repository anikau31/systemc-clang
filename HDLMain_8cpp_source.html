336 void HDLMain::SCport2hcode(ModuleInstance::portMapType pmap, hNode::hdlopsEnum h_op,
337 hNodep &h_info, hdecl_name_map_t &mod_vname_map) {
338 //clang::DiagnosticsEngine &diag_engine{getContext().getDiagnostics()};
339
340 for (ModuleInstance::portMapType::iterator mit = pmap.begin(); mit != pmap.end();
341 mit++) {
342 string objname = get<0>(*mit);
343
344 LLVM_DEBUG(llvm::dbgs() << "object name is " << objname << " and h_op is "
345 << h_op << "\n");
346
347 PortDecl *pd = get<1>(*mit);
348 Tree<TemplateType> *template_argtp =
349 (pd->getTemplateType())->getTemplateArgTreePtr();
350
351 std::vector<llvm::APInt> array_sizes = pd->getArraySizes();
352
353 HDLt.SCtype2hcode(objname, template_argtp, &array_sizes, h_op,
354 h_info); // passing the sigvarlist
355
356 // if this is a duplicate name due to inheritance
357 // create a new name and add it to the module level vname map
358 // this map will be passed to all calls to HDLBody to merge into
359 // its vname_map
360
361 if (module_vars.count(objname)) {
362 LLVM_DEBUG(llvm::dbgs() << "duplicate object " << objname << "\n");
363 NamedDecl * portdecl = pd->getAsVarDecl();
364 if (!portdecl)
365 portdecl = pd->getAsFieldDecl();
366 if (portdecl)
367 mod_vname_map.add_entry(portdecl, objname, h_info->child_list.back());
368 //string newn = mod_newn.newname();
369 //objname+="_var"+newn;
370 }
371 else module_vars.insert(objname);
372
373 // check for initializer
374 if (h_op == hNode::hdlopsEnum::hVardecl) {
375 VarDecl *vard = pd->getAsVarDecl();
376 if (vard) {
377 LLVM_DEBUG(llvm::dbgs() << "var decl dump follows\n");
378 LLVM_DEBUG(vard->dump(llvm::dbgs()));
379 if (vard->hasInit()) {
380 APValue *apval = vard->getEvaluatedValue();
381 if (apval && apval->isInt()) {
382 hNodep h_lit = new hNode((apval->getInt()).toString(10),
383 hNode::hdlopsEnum::hLiteral);
384 hNodep h_varinit = new hNode(hNode::hdlopsEnum::hVarInit);
385 h_varinit->child_list.push_back(h_lit);
386 (h_info->child_list.back())->child_list.push_back(h_varinit);
387 }
388 }
389 } else {
390 FieldDecl *fieldd = pd->getAsFieldDecl();
391 if (fieldd) {
392 LLVM_DEBUG(llvm::dbgs() << "field decl dump follows\n");
393 LLVM_DEBUG(fieldd->dump(llvm::dbgs()));
394 Expr* initializer = fieldd->getInClassInitializer();
395 if (initializer != NULL) {
396 LLVM_DEBUG(llvm::dbgs() << "field initializer dump follows\n");
397 LLVM_DEBUG(initializer->dump(llvm::dbgs(), getContext()));
398 hNodep h_init = new hNode(hNode::hdlopsEnum::hVarInit);
399 //HDLBody xmethod(initializer, h_init, main_diag_engine, getContext(), mod_vname_map);
400 xbodyp->Run(initializer, h_init, rnomode);
401 (h_info->child_list.back())->child_list.push_back(h_init);
402 }
403 }
404 }
405 }
406 }

514 void HDLMain::MakeResetMap( resetvar_map_t &threadresetmap, hNodep h_allsenslists)
515 {
516 // top node is a noop, then child list has the sensitivity lists:
517 //hNoop NONAME [
518 //hSenslist mc_proc [
519 //hSensvar NONAME [
520 //hVarref s_fp##valid NOLIST
521 //hNoop always NOLIST
522 // ]
523 // hSensvar NONAME [
524 // hVarref s_fp##data NOLIST
525 // hNoop always NOLIST
526 // ]
527 // or
528 //hSenslist break_in_for_wait0 [
529 // hSensvar ASYNC [
530 // hVarref arst NOLIST
531 // hLiteral 0 NOLIST
532 // ]
533 // ]
534
535 if (h_allsenslists != NULL) {
536 for (hNodep h_onesenslist : h_allsenslists->child_list) {
537 string threadname = h_onesenslist->getname();
538 for ( hNodep h_sensvar : h_onesenslist->child_list) {
539 if (h_sensvar->getname() == "NONAME") continue; // non-reset var has null hSensvar name
540 threadresetmap[threadname] = h_sensvar; // only one reset per thread
541 break;
542 }
543 }
544 }
