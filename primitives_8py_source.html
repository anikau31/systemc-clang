6class TypeContext(object):
7 def __init__(self, prefix=None, suffix=None, sep=None):
8 self.prefix = prefix
9 self.suffix = suffix
10 self.sep = sep
11
12 @staticmethod
13 def merge_field_default_this(this, that):
14 if this is None and that is not None:
15 return that
16 elif this is not None and that is None:
17 return this
18 elif this is None and that is None:
19 return None
20 else:
21 return that
22
23 def update_not_none(self, that):
24 """merges fields of two contexts, throws error if both fields are non None"""
25 if that is None:
26 return copy.copy(self)
27 prefix = TypeContext.merge_field_default_this(self.prefix, that.prefix)
28 suffix = TypeContext.merge_field_default_this(self.suffix, that.suffix)
29 sep = TypeContext.merge_field_default_this(self.sep, that.sep)
30 return TypeContext(prefix=prefix, suffix=suffix, sep=sep)
