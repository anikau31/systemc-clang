104 bool HDLBody::TraverseStmt(Stmt *stmt) {
105 LLVM_DEBUG(llvm::dbgs() << "In TraverseStmt\n");
106 if (stmt == nullptr) return true; // null statement, keep going
107
108 if (isa<CompoundStmt>(stmt)) {
109 LLVM_DEBUG(llvm::dbgs()
110 << "calling traverse compoundstmt from traversestmt\n");
111 TraverseCompoundStmt((CompoundStmt *)stmt);
112 } else if (isa<DeclStmt>(stmt)) {
113 TraverseDeclStmt((DeclStmt *)stmt);
114 } else if (isa<CallExpr>(stmt)) {
115 if (CXXOperatorCallExpr *opercall = dyn_cast<CXXOperatorCallExpr>(stmt)) {
116 LLVM_DEBUG(llvm::dbgs() << "found cxxoperatorcallexpr\n");
117 TraverseCXXOperatorCallExpr(opercall);
118 } else if (isa<CXXMemberCallExpr>(stmt)) {
119 TraverseCXXMemberCallExpr((CXXMemberCallExpr *)stmt);
120 } else {
121 TraverseCallExpr((CallExpr *)stmt);
122 }
123 } else if (isa<BinaryOperator>(stmt)) {
124 TraverseBinaryOperator((BinaryOperator *)stmt);
125 } else if (isa<UnaryOperator>(stmt)) {
126 TraverseUnaryOperator((UnaryOperator *)stmt);
127 } else if (isa<ConditionalOperator>(stmt)) {
128 TraverseConditionalOperator((ConditionalOperator *)stmt);
129 } else if (isa<MaterializeTemporaryExpr>(stmt)) {
130 TraverseStmt(((MaterializeTemporaryExpr *)stmt)->getSubExpr());
131 // TraverseStmt(((MaterializeTemporaryExpr *) stmt)->getTemporary());
132 } else if (isa<DeclRefExpr>(stmt)) {
133 TraverseDeclRefExpr((DeclRefExpr *)stmt);
134 } else if (isa<MemberExpr>(stmt)) {
135 TraverseMemberExpr((MemberExpr *)stmt);
136 } else if (isa<IntegerLiteral>(stmt)) {
137 TraverseIntegerLiteral((IntegerLiteral *)stmt);
138 } else if (isa<CXXBoolLiteralExpr>(stmt)) {
139 TraverseCXXBoolLiteralExpr((CXXBoolLiteralExpr *)stmt);
140 } else if (isa<IfStmt>(stmt)) {
141 LLVM_DEBUG(llvm::dbgs() << "Found if stmt\n");
142 TraverseIfStmt((IfStmt *)stmt);
143 } else if (isa<ForStmt>(stmt)) {
144 LLVM_DEBUG(llvm::dbgs() << "Found for stmt\n");
145 TraverseForStmt((ForStmt *)stmt);
146 } else if (isa<WhileStmt>(stmt)) {
147 LLVM_DEBUG(llvm::dbgs() << "Found while stmt\n");
148 TraverseWhileStmt((WhileStmt *)stmt);
149 } else if (isa<DoStmt>(stmt)) {
150 LLVM_DEBUG(llvm::dbgs() << "Found do-while stmt\n");
151 TraverseDoStmt((DoStmt *)stmt);
152 } else if (isa<SwitchStmt>(stmt)) {
153 LLVM_DEBUG(llvm::dbgs() << "Found switch stmt\n");
154 TraverseSwitchStmt((SwitchStmt *)stmt);
155 } else if (isa<CaseStmt>(stmt)) {
156 LLVM_DEBUG(llvm::dbgs() << "Found case stmt\n");
157 hNodep old_hret = h_ret;
158 hNodep hcasep = new hNode(hNode::hdlopsEnum::hSwitchCase);
159 if (ConstantExpr *expr =
160 dyn_cast<ConstantExpr>(((CaseStmt *)stmt)->getLHS())) {
161 llvm::APSInt val = expr->getResultAsAPSInt();
162 hcasep->child_list.push_back(
163 new hNode(systemc_clang::utils::apint::toString(val), hNode::hdlopsEnum::hLiteral));
164 }
165
166 TraverseStmt(((CaseStmt *)stmt)->getSubStmt());
167 if (h_ret != old_hret)
168 hcasep->child_list.push_back(h_ret);
169 else
170 hcasep->child_list.push_back(new hNode(hNode::hdlopsEnum::hUnimpl));
171
172 h_ret = hcasep;
173 } else if (isa<DefaultStmt>(stmt)) {
174 LLVM_DEBUG(llvm::dbgs() << "Found default stmt\n");
175 hNodep old_hret = h_ret;
176 hNodep hcasep = new hNode(hNode::hdlopsEnum::hSwitchDefault);
177 TraverseStmt(((DefaultStmt *)stmt)->getSubStmt());
178 if (h_ret != old_hret)
179 hcasep->child_list.push_back(h_ret);
180 else
181 hcasep->child_list.push_back(new hNode(hNode::hdlopsEnum::hUnimpl));
182 h_ret = hcasep;
183 } else if (isa<BreakStmt>(stmt)) {
184 //const unsigned cxx_record_id =
185 //diag_e.getCustomDiagID(clang::DiagnosticsEngine::Remark,
186 // "Break stmt not supported, substituting noop");
187 //clang::DiagnosticBuilder diag_builder{
188 //diag_e.Report(stmt->getBeginLoc(), cxx_record_id)};
189 LLVM_DEBUG(llvm::dbgs() << "Found break stmt\n");
190 h_ret = new hNode(thismode == rthread ? hNode::hdlopsEnum::hReturnStmt : hNode::hdlopsEnum::hBreak);
191 } else if (isa<ContinueStmt>(stmt)) {
192 LLVM_DEBUG(llvm::dbgs() << "Found continue stmt\n");
193 h_ret = new hNode(hNode::hdlopsEnum::hContinue);
194 } else if (isa<CXXDefaultArgExpr>(stmt)) {
195 TraverseStmt(((CXXDefaultArgExpr *)stmt)->getExpr());
196 } else if (isa<ReturnStmt>(stmt)) {
197 hNodep hretstmt = new hNode(hNode::hdlopsEnum::hReturnStmt);
198 if (((ReturnStmt *)stmt)->getRetValue() !=nullptr) {
199 TraverseStmt(((ReturnStmt *)stmt)->getRetValue());
200 hretstmt->child_list.push_back(h_ret);
201 }
202 h_ret = hretstmt;
203 } else if (isa<CXXTemporaryObjectExpr>(stmt)) {
204 int nargs = ((CXXTemporaryObjectExpr *)stmt)->getNumArgs();
205 if (nargs == 0) { // end of the road
206 h_ret = new hNode(
207 "0",
208 (hNode::hdlopsEnum::hLiteral)); // assume this is an initializer of 0
209 } else {
210 Expr **objargs = ((CXXTemporaryObjectExpr *)stmt)->getArgs();
211 for (int i = 0; i < nargs; i++) {
212 TraverseStmt(objargs[i]);
213 }
214 }
215 }
216 else if (isa<InitListExpr>(stmt)) {
217 hNodep h_initlist = new hNode(hNode::hdlopsEnum::hVarInitList);
218 for (auto tmpexpr: ((InitListExpr *) stmt)->inits()) {
219 TraverseStmt(tmpexpr);
220 h_initlist->append(h_ret);
221 }
222 h_ret = h_initlist;
223 }
224 else {
225 if (isa<CXXConstructExpr>(stmt)) {
226 CXXConstructExpr *exp = (CXXConstructExpr *)stmt;
227 // CXXConstructExpr argument not yet handled
228 // LLVM_DEBUG(llvm::dbgs()
229 // << "CXXConstructExpr found, expr below\n");
230 // LLVM_DEBUG(exp->dump(llvm::dbgs(), ast_context_));
231
232 if ((exp->getNumArgs() == 1) && (isa<IntegerLiteral>(exp->getArg(0)))) {
233 LLVM_DEBUG(llvm::dbgs()
234 << "CXXConstructExpr followed by integer literal found\n");
235 LLVM_DEBUG(exp->dump(llvm::dbgs(), ast_context_));
236 IntegerLiteral *lit = (IntegerLiteral *)exp->getArg(0);
237 string s = systemc_clang::utils::apint::toString(lit->getValue());
238 // need to add type to back of h_ret
239 FindTemplateTypes *te = new FindTemplateTypes();
240 te->Enumerate((exp->getType()).getTypePtr());
241 HDLType HDLt;
242 hNodep h_tmp = new hNode(hNode::hdlopsEnum::hNoop);
243 HDLt.SCtype2hcode(s, te->getTemplateArgTreePtr(), 0,
244 hNode::hdlopsEnum::hLiteral, h_tmp);
245 h_ret = h_tmp->child_list.back();
246 return true;
247 }
248 }
249 LLVM_DEBUG(llvm::dbgs()
250 << "stmt type " << stmt->getStmtClassName()
251 << " not recognized, calling default recursive ast visitor\n");
252 hNodep oldh_ret = h_ret;
253 RecursiveASTVisitor::TraverseStmt(stmt);
254 if (h_ret != oldh_ret) {
255 LLVM_DEBUG(
256 llvm::dbgs()
257 << "default recursive ast visitor called - returning translation\n");
258 return true;
259 }
260 for (auto arg : stmt->children()) {
261 LLVM_DEBUG(llvm::dbgs() << "child stmt type "
262 << ((Stmt *)arg)->getStmtClassName() << "\n");
263 if( isa<CXXThisExpr>(arg)) continue;
264 TraverseStmt(arg);
265 if (h_ret == oldh_ret) {
266 LLVM_DEBUG(llvm::dbgs() << "child stmt not handled\n");
267 // no output generated
268 h_ret = new hNode(arg->getStmtClassName(), hNode::hdlopsEnum::hUnimpl);
269 }
270 }
271 }
272
273 return true;
274 }

536 LLVM_DEBUG(expr->getDecl()->dump(llvm::dbgs()));
