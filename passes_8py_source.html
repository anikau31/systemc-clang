26 prev = tree

32 sc.visit(prev)
33 prev = LiteralExpansion(structure=sc.hier).visit(prev)
34 prev = SliceMerge().visit(prev)
35 prev = CommaTransformation().visit(prev)
36 f = TypeDefFilter()

38 prev = NodeMergePass().visit(prev)
39 prev = PortExpansion().visit(prev)
40 # note typedef should be after port expansion to prevent duplicate valid/ready
41 prev = TypedefExpansion(f.types).visit(prev)
42 prev = FunctionInfoPass().visit(prev)
43 prev = FunctionInfoPass2().visit(prev)
44 prev = FunctionParamMarker().visit(prev)
