336 bool HDLConstructorHcode::SetupSenslist(hNodep hp) {
337
338 // hMethodCall sc_coresc_simcontextcreate_method_process:create_method_process [
339 // hMethodCall sc_coresc_get_curr_simcontext:sc_get_curr_simcontext NOLIST
340 // hUnimpl StringLiteral NOLIST
341 // hLiteral 0 NOLIST
342 // hUnop & [
343 // hMethodCall zhwencode_blockfp_t11_52_2mc_proc:mc_proc NOLIST
344 // ]
345 // hUnimpl CXXThisExpr NOLIST
346 // hLiteral 0 NOLIST
347 // ]
348

350 for (hNodep hpi: hp->child_list) {
351 if (hpi->h_op == hNode::hdlopsEnum::hUnop) {
352 std::size_t found = (hpi->child_list[0]->h_name).find(qualnamedelim);

355 hNode::hdlopsEnum::hSenslist));

359 hnewsens.push_back(new hNode(hpi->child_list[0]->h_name, hNode::hdlopsEnum::hSenslist));
360 return true;
361 }
362 }
363 }
364 }

368 void HDLConstructorHcode::UnrollSensitem(hNodep &hp_orig, std::vector<for_info_t> &for_info) {
369 // hBinop << [
370 // hVarref sensitive NOLIST
371 // hNoop pos [
372 // hVarref clk NOLIST
373 // ]
374 // ]
375
376 // check for list of sens items
377 if (isInitSensitem(hp_orig->child_list[0])) {
378 UnrollSensitem(hp_orig->child_list[0], for_info);
379 }
380
381 // at a primitive sens item
382 hNodep hp = HnodeDeepCopy(hp_orig); // need to keep the subtrees when the original tree gets released
383
384 hp->h_op = hNode::hdlopsEnum::hSensvar;
385 hp->h_name = noname;
386
387 delete hp->child_list[0]; // release that hnode
388 hp->child_list.erase(hp->child_list.begin()); // remove the first item
389 if (!for_info.empty()) {
390 SubstituteIndex(hp, for_info);
391 }
392
393 // check for edge sensitivity
394 // eg
395 // hSensvar pos [ <-- hp
396 // hNoop pos [ <-- hedge
397 // hVarref clk NOLIST <-- hedge->child_list[0]
398 // ]
399 // ]
400
401 if (isEdge(hp->child_list[0]->h_name)) {
402 hNodep hedge = hp->child_list[0];
403 hp->child_list[0] = hedge->child_list[0];
404 hedge->child_list.pop_back();
405 hp->child_list.push_back(hedge);
406 }
407 else {
408
409 if (isSimEvent(hp->child_list[0]->h_name)) {
410
411 // hSensvar NONAME [
412 // hNoop value_changed_event [
413 // hVarref c_fp##ready NOLIST
414 // ]
415 // hNoop always NOLIST
416 // ]
417
418 hNodep htmp = hp->child_list[0]->child_list[0];
419 hp->child_list.erase(hp->child_list.begin());
420 hp->child_list.push_back(htmp);
421 }
422
423 hp->child_list.push_back(new hNode("always", hNode::hdlopsEnum::hNoop));
424 };
425
426 hnewsens.back()->child_list.push_back(hp);

429 void HDLConstructorHcode::HDLLoop(hNodep &hp, std::vector<for_info_t> &for_info ) {
430 // check in order of expected frequency
431 if (isInitPB(hp)) {
432 UnrollBinding(hp, for_info);
433 }
434 else if (isInitSensitem(hp)) {
435 UnrollSensitem(hp, for_info);
436 }
437
438 else if ((hp->h_op == hNode::hdlopsEnum::hForStmt) && (hp->child_list.size() > 3)) {
439 PushRange(hp, for_info); // fill in name, lo, hi, step
440 for (int forloopix = for_info.back().lo; forloopix < for_info.back().hi; forloopix+=for_info.back().step) {
441 for_info.back().curix = forloopix;
442 for (int i=3; i<hp->child_list.size(); i++) {
443 if (isInitPB(hp->child_list[i])) {// hcode indicating port binding
444 UnrollBinding(hp->child_list[i], for_info); // unroll all bindings in this range
445 }
446 else if (isInitSensitem(hp->child_list[i])) { // hcode indicating sensitivity item
447 UnrollSensitem(hp->child_list[i], for_info); // unroll all sensitems in this range
448 }
449 else if ((hp->child_list[i]->h_op == hNode::hdlopsEnum::hForStmt) ||
450 (hp->child_list[i]->h_op == hNode::hdlopsEnum::hCStmt))
451 HDLLoop(hp->child_list[i], for_info);
452 }
453 }
454 for_info.pop_back();
455 }
456 else if (hp->h_op == hNode::hdlopsEnum::hCStmt) {
457 for (hNodep hpc:hp->child_list) {
458 HDLLoop(hpc, for_info);
459 }
460 }
461 else if (isMethodCall(hp)) { // hVarAssign child[1] is a method call;
462 if (!SetupSenslist(hp->child_list[1])) { // points to first hMethodCall, push SC_METHOD name onto hnewsens
463 // oops couldn't parse it
464 hnewsens.push_back(new hNode( "METHOD ???", hNode::hdlopsEnum::hSenslist));
465 }
466 }
467 else {
468 // check for thread sensitivity declarations
469 int threadsensitem = isThreadSensitem(hp);
470 if (threadsensitem >0 ) {
471 // e.g. hMethodCall sc_core__sc_module__async_reset_signal_is:async_reset_signal_is [
472 // hVarref reset NOLIST
473 // hLiteral 0 NOLIST
474 // ]
475 LLVM_DEBUG(llvm::dbgs() << "HDLHNode: found thread sens item " << "\n");
476 hNodep hpsens = HnodeDeepCopy(hp); // need to keep the subtrees when the original tree gets released
477
478 hpsens->set(hNode::hdlopsEnum::hSensvar, threadsensitem == reset_async? "ASYNC": "SYNC");
479 if (hnewsens.size()==0) // this shouldn't be the case, but whatever
480 hnewsens.push_back(new hNode( "METHOD ???", hNode::hdlopsEnum::hSenslist));
481 hnewsens.back()->append(hpsens);
482 }
483 }
