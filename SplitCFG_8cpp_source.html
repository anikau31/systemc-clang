124const SplitCFG::SplitCFGPath SplitCFG::dfs_visit_wait(
125 const SplitCFGBlock* BB,
126 llvm::SmallPtrSet<const SplitCFGBlock*, 32>& visited_blocks,
127 llvm::SmallVectorImpl<const SplitCFGBlock*>& waits_to_visit,
128 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_waits,
129 llvm::SmallVector<SplitCFGPathPair>& curr_path) {
130 bool true_path_{false};
131 bool false_path_{false};
132 bool stop_local_path{false};
133
134 SplitCFGPath local_path;
136 if (BB->succ_empty()) {
137 return local_path;
138 }
139
140 popping_ = false;
141 // successors to visit
142 llvm::SmallVector<
143 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>, 8>
144 to_visit;
145
146 visited_blocks.insert(BB);
147 to_visit.push_back(std::make_pair(BB, BB->succ_begin()));
148
149 // Loop
150 do {
151 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>& Top =
152 to_visit.back();
153 const SplitCFGBlock* ParentBB{Top.first};
154 SplitCFGBlock::const_succ_iterator& I{Top.second};
155
157 bool bb_has_wait{(ParentBB->hasWait())};
158 bool cond_block{isConditional(ParentBB)};
159
161 int id{-1};
162 if (!popping_) {
163 llvm::dbgs() << "Visit Parent BB# " << ParentBB->getBlockID() << " ; ";
164 SupplementaryInfo info{ParentBB};
165 info.path_idx_ = curr_path.size();
166 curr_path.push_back(std::make_pair(ParentBB, info));
167 id = curr_path.size() - 1;
168
169 if (!stop_local_path) {
170 local_path.push_back(
171 std::make_pair(ParentBB, SupplementaryInfo{ParentBB}));
172 }
173 } else {
174 // llvm::dbgs() << "POPPING\n";
175 }
178 if (bb_has_wait) {
179 llvm::dbgs() << " has WAIT ";
180 bool found_suc{getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB)};
181 // A wait to be visited is added.
182 if (visited_waits.insert(ParentBB).second) {
183 waits_to_visit.insert(waits_to_visit.begin(), BB);
184 ++next_state_count_;
185 auto wit{wait_next_state_.find(ParentBB)};
186 if (wit == wait_next_state_.end()) {
187 wait_next_state_.insert(
188 std::make_pair(ParentBB, std::make_pair(BB, next_state_count_)));
189 }
190 }
191 }
192
193 // If there is a successor that has not been visited, then remember that
194 // block.
195 llvm::SmallPtrSet<const SplitCFGBlock*, 32> save_visited_blocks{
196 visited_blocks};
197 // dumpVisitedBlocks(visited_blocks);
198
199 bool found_succ{getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB)};
200 if (found_succ) {
201 llvm::dbgs() << "Found successor BB " << BB->getBlockID()
202 << " for parentBB " << ParentBB->getBlockID() << "\n";
203 }
204
205 // Use the recursive call for loops with 2 successors, and IF blocks.
206 // START: LOOP/SPECIAL NODE
207 if ((isLoopWithTwoSuccessors(ParentBB) || isConditional(ParentBB)) &&
208 found_succ) {
209 do {
210 // llvm::dbgs() << "\n==============================================";
211 llvm::dbgs() << "\n#### ParentBB " << ParentBB->getBlockID()
212 << " is a loop with 2 succ or conditional. Succ BB# "
213 << BB->getBlockID() << "\n";
214
218 auto block_path{path_info_.find(ParentBB)};
219 if (block_path != path_info_.end()) {
220 if (const auto true_block_succ = *ParentBB->succ_begin()) {
221 if (true_block_succ == BB) {
222 true_path_ = true;
223 llvm::dbgs() << "Going down the TRUE path for BB"
224 << block_path->first->getBlockID() << " \n";
225
226 } else {
227 false_path_ = true;
228 llvm::dbgs() << "Going down the FALSE path for BB"
229 << block_path->first->getBlockID() << " \n";
230 }
231 }
232 }
233
234 llvm::SmallPtrSet<const SplitCFGBlock*, 32> loop_visited_blocks{
235 save_visited_blocks};
236 // llvm::SmallPtrSet<const SplitCFGBlock*, 32> loop_visited_blocks{};
237
238 // TODO: Do we need this?
239 // visited_blocks.insert(BB);
240
241 // ParentBB has been visited so don't revisit it
242 loop_visited_blocks.insert(ParentBB);
243
244 llvm::dbgs() << "\n==============================================";
245 llvm::dbgs() << "\n[START] Recursive DFS starting at BB "
246 << BB->getBlockID() << " parent BB# "
247 << ParentBB->getBlockID() << " T " << true_path_ << " F "
248 << false_path_ << "\n";
249 llvm::dbgs() << "loop visited ";
250 dumpVisitedBlocks(loop_visited_blocks);
251 llvm::dbgs() << "visited ";
252 dumpVisitedBlocks(visited_blocks);
253 llvm::dbgs() << "local_path : ";
254 dumpSmallVector(local_path);
255 llvm::dbgs() << "\n";
256
257 SplitCFGPath sub_path_to_special_node;
258 // =================== DFS Recurse ====================
259 sub_path_to_special_node = dfs_visit_wait(
260 BB, loop_visited_blocks, waits_to_visit, visited_waits, curr_path);
261 llvm::dbgs() << "\n";
262 // =================== END DFS Recurse ====================
263
264 addPathToSpecialNode(sub_path_to_special_node);
265
268 // updateVisitedBlocks(loop_visited_blocks, loop_visited_blocks);
269 llvm::dbgs() << "[AFTER] DFS for BB# " << BB->getBlockID()
270 << " for parent BB# " << ParentBB->getBlockID() << " T "
271 << true_path_ << " F " << false_path_ << "\n";
272 llvm::dbgs() << "loop visited ";
273 dumpVisitedBlocks(loop_visited_blocks);
274 llvm::dbgs() << "visited ";
275 dumpVisitedBlocks(visited_blocks);
276 llvm::dbgs() << "sub_path : ";
277 dumpSmallVector(sub_path_to_special_node);
278
279 llvm::dbgs() << "\n[END] Recurse DFS at BB " << BB->getBlockID()
280 << " parent BB# " << ParentBB->getBlockID() << " T "
281 << true_path_ << " F " << false_path_ << "\n";
282
283 // std::cin.get();
286 // new_visited.insert(BB);
287
288 found_succ = getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB);
289
290 if (true_path_) {
291 llvm::dbgs() << "[TRUE PATH] Set path for BB "
292 << ParentBB->getBlockID() << " to sub_path ";
293 dumpSmallVector(sub_path_to_special_node);
294 llvm::dbgs() << "\n";
295
296 true_path_ = false;
297 auto& info{curr_path[id]};
298 setTruePathInfo(ParentBB, sub_path_to_special_node, curr_path.size());
299 llvm::dbgs() << "curr_path :";
300 dumpSmallVector(curr_path);
301 info.second.false_idx_ = curr_path.size();
302
303 // If there is no successor, then we should find the next block on the
304 // false path and set that to the false_idx_ from curr_path.
305 if (!found_succ) {
306 // There is no successor for ParentBB that is not visited. So, it
307 // must have already been visited, and we must find it in the
308 // curr_path.
309 const clang::CFGBlock* cblock{ParentBB->getCFGBlock()};
310 const clang::CFGBlock* fsucc{*cblock->succ_rbegin()};
311
312 if (fsucc) {
313 fsucc->dump();
314 // Find the index from curr_path, and set it to the false_idx_, if
315 // found.
316 for (auto const visited_blk : curr_path) {
317 if (visited_blk.first->getCFGBlock() == fsucc) {
318 info.second.false_idx_ = visited_blk.second.path_idx_;
319 }
320 }
321 }
322 }
323
324 // llvm::dbgs() << "ID is " << id << " INFO id is "
325 // << info.second.path_idx_ << " SB "
326 // << info.first->getBlockID() << " size of path "
327 // << curr_path.size() << "\n";
328 } else if (false_path_) {
329 // llvm::dbgs() << "FALSE PATH DIFF BB# " << ParentBB->getBlockID();
330 llvm::dbgs() << "[FALSE PATH] Set path for BB "
331 << ParentBB->getBlockID() << " to sub_path ";
332 dumpSmallVector(sub_path_to_special_node);
333 llvm::dbgs() << "\n";
334 false_path_ = false;
335 setFalsePathInfo(ParentBB, sub_path_to_special_node);
336 dumpPathInfo();
337 dumpAllPathInfo();
338 }
339
340 // dumpVisitedBlocks(new_visited);
341
342 llvm::dbgs() << "\n==============================================";
343
352 //
353
354 if (!found_succ) {
355 llvm::dbgs() << "\nNO SUCCESOR for BB " << ParentBB->getBlockID()
356 << "\n";
357 }
360 if (!found_succ) {
361 updateVisitedBlocks(visited_blocks, loop_visited_blocks);
362 }
363
364 } while (found_succ);
365
366 llvm::dbgs() << "@@@@@@ Exiting loop\n";
367 dumpSmallVector(sub_path_to_special_node_);
368
369 } else {
370 // Recursive call in the if-then will traverse the subgraph.
371 // Only insert successor if recursive call does not visit subgraph.
372 }
373 addSuccessorToVisitOrPop(bb_has_wait, BB, to_visit, found_succ);
374 // std::cin.get();
375 // llvm::dbgs() << "to_visit ";
376 // dumpSmallVector(to_visit);
377 // llvm::dbgs() << " End loop \n";
378 } while (!to_visit.empty());
379 return local_path;
380}

639void SplitCFG::splitBlock(clang::CFGBlock* block) {
640 assert(block != nullptr);
641
642 unsigned int num_elements{block->size()};
643
644 llvm::dbgs() << "\n\nChecking if block " << block->getBlockID()
645 << " has a wait()\n";
646 llvm::dbgs() << "Number of elements " << num_elements << "\n";
647 // We are going to have two vectors.
648 // 1. A vector of vector pointers to CFGElements.
649 // 2. A vector of pointers to CFGElements that are waits.
650 //
651
652 VectorCFGElementPtr vec_elements{};
653 llvm::SmallVector<std::pair<VectorCFGElementPtr, bool>> split_elements;
654
656 if (num_elements == 0) {
657 split_elements.push_back(std::make_pair(vec_elements, false));
658 } else {
659 bool has_wait{false};
660 for (auto const& element : block->refs()) {
661 has_wait = false;
662 llvm::dbgs() << "Is this a wait? ";
663 element->dump();
664
668 const clang::CFGElement* element_ptr{element.operator->()};
670 if (isElementWait(*element)) {
671 llvm::dbgs() << " IT IS A WAIT\n";
673 if (num_elements == 1) {
674 llvm::dbgs() << "DBG: Only one statement and it is a wait().\n";
675 }
676
677 // Add all elements before the wait into the split elements.
678 if (vec_elements.size() != 0) {
679 split_elements.push_back(std::make_pair(vec_elements, false));
680 vec_elements.clear();
681 }
682
684 vec_elements.push_back(element_ptr);
685 split_elements.push_back(std::make_pair(vec_elements, true));
686 vec_elements.clear();
687
688 has_wait = true;
689 } else {
690 vec_elements.push_back(element_ptr);
691 }
692 }
693
694 if (vec_elements.size() != 0) {
695 split_elements.push_back(std::make_pair(vec_elements, has_wait));
696 }
697 }
698
699 createWaitSplitCFGBlocks(block, split_elements);
700}

702void SplitCFG::createWaitSplitCFGBlocks(
703 clang::CFGBlock* block,
704 const llvm::SmallVectorImpl<std::pair<SplitCFG::VectorCFGElementPtr, bool>>&
705 split_elements) {
706 // dumpSplitElements(split_elements);
708 llvm::dbgs() << "Number of entries in split_elements "
709 << split_elements.size() << "\n";
710 unsigned int id{0};
711 SplitCFGBlock* prev_block{nullptr};
712
713 for (auto const& elements : split_elements) {
714 llvm::dbgs() << "Element number " << id << " ";
716 SplitCFGBlock* new_split{nullptr};
717
718 bool already_exists{false};
719 // id = 0 is when it is the first sequence of elements.
720 if (id == 0) {
721 auto scit{sccfg_.find(block->getBlockID())};
722 // Must be true.
723 // bool already_exists{scit != sccfg_.end()};
724 new_split = scit->second;
725 new_split->id_ = block->getBlockID();
726
727 // Successor: succ block from CFGBlock (overridden)
729 if (split_elements.size() == 1) {
730 addSuccessors(new_split, block);
731 }
732 // Prececessor: prev block from CFGBlock
733 addPredecessors(new_split, block);
734 } else {
735 auto scit{sccfg_.find(block->getBlockID())};
736 new_split = new SplitCFGBlock{}; //*scit->second};
737 new_split->id_ = block->getBlockID() * 10 + id;
738
739 auto is_wait = elements.second;
740 auto stmt{block->getTerminatorStmt()};
741 // A wait block cannot be a conditional.
742 new_split->is_conditional_ = !is_wait && (
743 stmt && (llvm::isa<clang::IfStmt>(stmt) ||
744 llvm::isa<clang::ConditionalOperator>(stmt)));
745
747 //

758 new_split->predecessors_.push_back(prev_block);
759 sccfg_.insert(std::make_pair(block->getBlockID() * 10 + id, new_split));
760 }

762 new_split->block_ = block;
763 new_split->has_wait_ = elements.second;
764 new_split->elements_ = elements.first;

771 }

778 }

780 prev_block = new_split;
781 ++id;

945void SplitCFG::createUnsplitBlocks() {
946 for (auto begin_it = cfg_->nodes_begin(); begin_it != cfg_->nodes_end();
947 ++begin_it) {
948 auto block{*begin_it};
949 SplitCFGBlock* new_block{new SplitCFGBlock{}};
950 new_block->id_ = block->getBlockID();
951
953 auto stmt{block->getTerminatorStmt()};
954 new_block->is_conditional_ =
955 stmt && (llvm::isa<clang::IfStmt>(stmt) ||
956 llvm::isa<clang::ConditionalOperator>(stmt));
957
958 bool last_succ_is_null{false};
959 if (block->succ_size() == 2) {
960 // Check that the last one is not NULL
961 if (*block->succ_rbegin() == nullptr) {
962 last_succ_is_null = true;
963 }
964 }
965
967 bool is_loop{stmt && (llvm::isa<clang::WhileStmt>(stmt) ||
968 llvm::isa<clang::ForStmt>(stmt) ||
969 llvm::isa<clang::DoStmt>(stmt))};
970
971 new_block->is_loop_with_two_succ_ =
972 (stmt && is_loop && (last_succ_is_null == false));
973
974 sccfg_.insert(std::make_pair(new_block->id_, new_block));
975 }
976
977 llvm::dbgs() << "sccfg( " << sccfg_.size() << ") ids: ";
978 for (auto const& entry : sccfg_) {
979 llvm::dbgs() << entry.first << " ";
980 }
981 llvm::dbgs() << "\n";
