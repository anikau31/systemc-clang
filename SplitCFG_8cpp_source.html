26void SplitCFG::dfs_pop_on_wait(
27 const SplitCFGBlock* basic_block,
28 llvm::SmallVectorImpl<const SplitCFGBlock*>& waits_in_stack,
29 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_waits) {
31 if (basic_block->succ_empty()) {
32 return;
33 }
34
35 llvm::SmallPtrSet<const SplitCFGBlock*, 8> visited;
36 llvm::SmallVector<
37 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>, 8>
38 visit_stack;
39 llvm::SmallVector<const SplitCFGBlock*, 8> in_stack;
40
41 visited.insert(basic_block);
42 visit_stack.push_back(std::make_pair(basic_block, basic_block->succ_begin()));
43
44 // llvm::SmallVector<const SplitCFGBlock*> curr_path;
45 VectorSplitCFGBlock curr_path;
46 do {
47 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>& Top =
48 visit_stack.back();
49 const SplitCFGBlock* parent_bb = Top.first;
50 SplitCFGBlock::const_succ_iterator& I = Top.second;
51
53 bool bb_has_wait{(parent_bb->hasWait())};
54
55 if (bb_has_wait) {
56 llvm::dbgs() << "BB# " << parent_bb->getBlockID()
57 << " has a wait in it\n";
58 llvm::dbgs() << "Visited BB#" << parent_bb->getBlockID() << "\n";
59 curr_path.push_back(parent_bb);
60
63 if (visited_waits.insert(parent_bb).second == true) {
64 // Insert the successor of the block.
65
66 while (I != parent_bb->succ_end()) {
67 basic_block = *I++;
68 if ((basic_block != nullptr)) {
69 llvm::dbgs() << "Insert successor of BB#" << parent_bb->getBlockID()
70 << ": BB#" << basic_block->getBlockID() << "\n";
71 // Mimic FIFO
72 waits_in_stack.insert(waits_in_stack.begin(), basic_block);
73 auto wit = wait_next_state_.find(parent_bb);
74 if (wit == wait_next_state_.end()) {
75 wait_next_state_.insert(std::make_pair(
76 parent_bb, std::make_pair(basic_block, next_state_count_)));
77 ++next_state_count_;
78 }
79 break;
80 }
81 }
82 }
83 }
84
87 bool FoundNew = false;
88 while (I != parent_bb->succ_end()) {
89 basic_block = *I++;
90 if ((basic_block != nullptr) && (!bb_has_wait) &&
91 (visited.insert(basic_block).second)) {
92 FoundNew = true;
93 break;
94 }
95 }
96
97 if (FoundNew) {
98 // Go down one level if there is a unvisited successor.
99 llvm::dbgs() << "Visited BB#" << parent_bb->getBlockID() << "\n";
100 visit_stack.push_back(
101 std::make_pair(basic_block, basic_block->succ_begin()));
102 curr_path.push_back(parent_bb);
103 } else {
104 // Go up one level.
105 // llvm::dbgs() << "pop: " << visit_stack.size() << "\n";
106 visit_stack.pop_back();
107 }
108 } while (!visit_stack.empty());
109
111 paths_found_.push_back(curr_path);
112}

114llvm::APInt SplitCFG::getWaitArgument(const clang::CFGElement& element) const {
115 if (auto cfg_stmt = element.getAs<clang::CFGStmt>()) {
116 auto stmt{cfg_stmt->getStmt()};
117
118 if (auto* expr = llvm::dyn_cast<clang::Expr>(stmt)) {
119 if (auto cxx_me = llvm::dyn_cast<clang::CXXMemberCallExpr>(expr)) {
120 auto args{cxx_me->getNumArgs()};
121 if (args == 1) {
122 if (auto first_arg{cxx_me->getArg(0)}) {
123 llvm::dbgs() << "*************** FIRST ARG ****************\n";
124 first_arg->dump();
125 const clang::IntegerLiteral* IntLiteral =
126 clang::dyn_cast<clang::IntegerLiteral>(first_arg);
127 llvm::dbgs() << " ARG VAL: " << IntLiteral->getValue() << "\n";
128 return IntLiteral->getValue();
129 }
130 }
131 }
132 }
133 }
134
135 return llvm::APInt{32, 0, false};

139 if (auto cfg_stmt = element.getAs<clang::CFGStmt>()) {

141
142 // stmt->dump();

144 if (auto cxx_me = llvm::dyn_cast<clang::CXXMemberCallExpr>(expr)) {

154 return false;

161 }
162 }

167void SplitCFG::splitBlock(clang::CFGBlock* block) {
168 assert(block != nullptr);
169
170 unsigned int num_elements{block->size()};
171
172 llvm::dbgs() << "\n\nChecking if block " << block->getBlockID()
173 << " has a wait()\n";
174 llvm::dbgs() << "Number of elements " << num_elements << "\n";
175 // We are going to have two vectors.
176 // 1. A vector of vector pointers to CFGElements.
177 // 2. A vector of pointers to CFGElements that are waits.
178 //
179
180 VectorCFGElementPtr vec_elements{};
181 llvm::SmallVector<std::pair<VectorCFGElementPtr, bool> > split_elements;
182
184 if (num_elements == 0) {
185 split_elements.push_back(std::make_pair(vec_elements, false));
186 } else {
187 bool has_wait{false};
188 for (auto const& element : block->refs()) {
189 has_wait = false;
190 llvm::dbgs() << "Is this a wait? ";
191 element->dump();
192
196 const clang::CFGElement* element_ptr{element.operator->()};
198 if (isElementWait(*element)) {
199 llvm::dbgs() << " IT IS A WAIT\n";
201 if (num_elements == 1) {
202 llvm::dbgs() << "DBG: Only one statement and it is a wait().\n";
203 }
204
205 // Add all elements before the wait into the split elements.
206 if (vec_elements.size() != 0) {
207 split_elements.push_back(std::make_pair(vec_elements, false));
208 vec_elements.clear();
209 }
210
212 vec_elements.push_back(element_ptr);
213 split_elements.push_back(std::make_pair(vec_elements, true));
214 vec_elements.clear();
215
216 has_wait = true;
217 } else {
218 vec_elements.push_back(element_ptr);
219 }
220 }
221
222 if (vec_elements.size() != 0) {
223 split_elements.push_back(std::make_pair(vec_elements, has_wait));
224 }
225 }
226
227 createWaitSplitCFGBlocks(block, split_elements);
228}

230void SplitCFG::createWaitSplitCFGBlocks(
231 clang::CFGBlock* block,
232 const llvm::SmallVectorImpl<
233 std::pair<SplitCFG::VectorCFGElementPtr, bool> >& split_elements) {
234 // dumpSplitElements(split_elements);
236 llvm::dbgs() << "Number of entries in split_elements "
237 << split_elements.size() << "\n";
238 unsigned int id{0};
239 SplitCFGBlock* prev_block{nullptr};
240
241 for (auto const& elements : split_elements) {
242 llvm::dbgs() << "Element number " << id << " ";
244 SplitCFGBlock* new_split{nullptr};
245
246 bool already_exists{false};
247 // id = 0 is when it is the first sequence of elements.
248 if (id == 0) {
249 auto scit{sccfg_.find(block->getBlockID())};
250 // Must be true.
251 // bool already_exists{scit != sccfg_.end()};
252 new_split = scit->second;
253 new_split->id_ = block->getBlockID();
254
255 // Successor: succ block from CFGBlock (overridden)
257 if (split_elements.size() == 1) {
258 addSuccessors(new_split, block);
259 }
260 // Prececessor: prev block from CFGBlock
261 addPredecessors(new_split, block);
262 } else {
263 new_split = new SplitCFGBlock{};
264 new_split->id_ = block->getBlockID() * 10 + id;
265
267 //
269 llvm::dbgs() << "SB# " << new_split->id_ << " size "
270 << split_elements.size() << "\n";
271 if (id == split_elements.size() - 1) {
272 llvm::dbgs() << " add successors of BB# " << block->getBlockID()
273 << "\n";
274 addSuccessors(new_split, block);
275 }
276 prev_block->successors_.push_back(new_split);
278 new_split->predecessors_.push_back(prev_block);
279 sccfg_.insert(std::make_pair(block->getBlockID() * 10 + id, new_split));
280 }
281
282 new_split->block_ = block;
283 new_split->has_wait_ = elements.second;
284 new_split->elements_ = elements.first;
285
286 // If it's a wait() then get the argument.
287 if (new_split->has_wait_) {
289 const clang::CFGElement* wait_el = (*new_split->elements_.begin());
290 new_split->wait_arg_ = getWaitArgument(*wait_el);
291 }
292
293 // Set the successor of new_split.
294 prev_block = new_split;
295 ++id;
296 }

347void SplitCFG::generate_paths() {
349 llvm::SmallPtrSet<const SplitCFGBlock*, 8> visited_waits;
350 VectorSplitCFGBlock waits_in_stack;
351
353 const clang::CFGBlock* block{&cfg_->getEntry()};
354 const SplitCFGBlock* entry{sccfg_[block->getBlockID()]};
355 // Mimic FIFO
356 waits_in_stack.insert(waits_in_stack.begin(), entry);
357 // Entry=>Entry is the reset.
358 wait_next_state_.insert(
359 std::make_pair(entry, std::make_pair(entry, next_state_count_)));
360 ++next_state_count_; // Reset has been assigned to 0
361 do {
362 entry = waits_in_stack.pop_back_val();
363 llvm::dbgs() << "Processing SB " << entry->getBlockID() << "\n";
364 dfs_pop_on_wait(entry, waits_in_stack, visited_waits);
365 llvm::dbgs() << "\n";
366 } while (!waits_in_stack.empty());
367
368 addNextStatesToBlocks();
369 dumpWaitNextStates();
370 dumpPaths();
371}

434void SplitCFG::construct_sccfg(const clang::CXXMethodDecl* method) {
435 cfg_ = clang::CFG::buildCFG(method, method->getBody(), &context_,
436 clang::CFG::BuildOptions());
437
438 clang::LangOptions lang_opts;
439 cfg_->dump(lang_opts, true);
440
441 createUnsplitBlocks();
442
443 for (auto begin_it = cfg_->nodes_begin(); begin_it != cfg_->nodes_end();
444 ++begin_it) {
445 auto block{*begin_it};
446
447 splitBlock(block);
448 }
449 llvm::dbgs() << "\nPrepare to update successors\n";
450 dumpSCCFG();
