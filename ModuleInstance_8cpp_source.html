127ModuleInstance::~ModuleInstance() {
128 DEBUG_WITH_TYPE("DebugDestructors", llvm::dbgs() << "\n~ModuleInstance\n";);
129 class_decl_ = nullptr;
130 constructor_stmt_ = nullptr;
131 instance_decl_ = nullptr;
132 // IMPORTANT: Only the instance-specific details should be deleted.
133 // DO NOT delete the information collected through incomplete types.
134 //
135
136 for (auto base : base_instances_) {
137 delete base;
138 }
139
140 for (auto &v : vef_) {
141 if (v != nullptr) {
142 delete v;
143 }
144 v = nullptr;
145 }
146 // Delete all pointers in ports.
147 for (auto &input_port : in_ports_) {
148 // It is a tuple
149 // 0. string, 1. PortDecl*
150 // Only one to delete is (1)
151 //
152 // delete input_port.second;
153 auto iport{get<1>(input_port)};
154 if (iport) {
155 delete iport;
156 }
157 }
158 in_ports_.clear();
159
160 for (auto &output_port : out_ports_) {
161 delete get<1>(output_port);
162 }
163 out_ports_.clear();
164
165 for (auto &io_port : inout_ports_) {
166 // Second is the PortDecl*.
167 delete get<1>(io_port);
168 }
169 inout_ports_.clear();
170
171 for (auto &other : other_fields_) {
172 // Second is the PortDecl*.
173 delete get<1>(other);
174 }
175 other_fields_.clear();
176 // Delete EntryFunction container
177 for (auto &ef : vef_) {
178 delete ef;
179 }
180 vef_.clear();
181
182 for (auto &sig : signals_) {
183 delete sig.second;
184 }

230void ModuleInstance::addPorts(const ModuleInstance::PortType &found_ports,
231 const std::string &port_type) {
232 if (port_type == "sc_in") {
233 std::copy(begin(found_ports), end(found_ports), back_inserter(in_ports_));
234 }
235 if (port_type == "sc_out") {
236 std::copy(begin(found_ports), end(found_ports), back_inserter(out_ports_));
237 }
238 if (port_type == "sc_inout") {
239 std::copy(begin(found_ports), end(found_ports),
240 back_inserter(inout_ports_));
241 }
242 if (port_type == "others") {
243 std::copy(begin(found_ports), end(found_ports),
244 back_inserter(other_fields_));
245 }
246
247 if (port_type == "sc_signal") {
249 for (auto const &signal_port : found_ports) {
250 auto port_decl{get<1>(signal_port)};
251 auto name{port_decl->getName()};
252
253 // SignalContainer
254 SignalDecl *signal_entry{new SignalDecl{*port_decl}};
255 signals_.insert(ModuleInstance::signalPairType(name, signal_entry));
256 }
257 }
258
259 if (port_type == "sc_stream_in") {
260 std::copy(begin(found_ports), end(found_ports),
261 back_inserter(istreamports_));
262 }
263 if (port_type == "sc_stream_out") {
264 std::copy(begin(found_ports), end(found_ports),
265 back_inserter(ostreamports_));
266 }

314void ModuleInstance::addProcess(
315 FindEntryFunctions::entryFunctionVectorType *efv) {
316 vef_ = *efv;
317 for (unsigned int i = 0; i < efv->size(); ++i) {
318 EntryFunctionContainer *ef = (*efv)[i];
319
320 // Set the entry name.
321 std::string entryName{ef->getName()};
322 std::string entryType{""};
323
324 // Set the process type
325 switch (ef->getProcessType()) {
326 case PROCESS_TYPE::THREAD: {
327 entryType = "SC_THREAD";
328 break;
329 }
330 case PROCESS_TYPE::METHOD: {
331 entryType = "SC_METHOD";
332 break;
333 }
334 case PROCESS_TYPE::CTHREAD: {
335 entryType = "SC_CTHREAD";
336 break;
337 }
338 default: {
339 entryType = "ERROR";
340 break;
341 }
342 }
343 process_map_.insert(processPairType(
344 entryName,
345 new ProcessDecl(entryType, entryName, ef->getEntryMethod(), ef)));
346 }
