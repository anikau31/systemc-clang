23 void registerMatchers(MatchFinder &finder) {
24 // Overview of the matcher
25 // Terminology: template parameter refers to the template typename name, and
26 // template arguments refers to the substituted type. For example, template
27 // <typename T> X{}; X<int> x; has the parameter as T and the argument as
28 // int.
29 //
30 // 1. Start matching within a ClassTemplateDecl. Note that the match() call
31 // has to be updated to ensure that the appropriate ClassTemplateDecl is
32 // provided.
33 // 2. The isExpansionInMainFile() makes the matcher focus on the provided
34 // file (input), and not the underlying C++ files.
35 //
36 // 3. We check if the ClassTemplateDecl has a CXXRecordDecl within it (note
37 // that it will have a CXXRecordDecl and a ClassTemplateSpecializationDecl,
38 // but we want the former to extract the template parameters).
39 // 4. For every fieldDecl that is a descendant, it is going to check if the
40 // type for the FieldDecl's desugared type is a TemplateSpecialization or a
41 // TemplateTypeParm. This should cover typedefs as well. An alternative way
42 // would be to just match with Type, and then in the callback one could cast
43 // it to respective types that one cares about.
44 //
45 /* clang-format off */
46
47 auto match_template_decl =
48 classTemplateDecl(
49 has(cxxRecordDecl(forEachDescendant(
50 fieldDecl(
51 anyOf(
52 hasType(hasUnqualifiedDesugaredType(recordType().bind("record_type")))
53 , hasType(hasUnqualifiedDesugaredType(templateSpecializationType().bind("specialization_type")))
54 , hasType(hasUnqualifiedDesugaredType(templateTypeParmType().bind("parm_type")))
55 , hasType(hasUnqualifiedDesugaredType(builtinType().bind( "builtin_type")))
56 ) // anyOf
57 ).bind("fd"))
58 )
59 )
60 ).bind("template_decl");
61
62 /* clang-format on */
63
64 finder.addMatcher(match_template_decl, this);
65 }
