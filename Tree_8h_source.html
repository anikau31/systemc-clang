203 std::string dft(TreeNodePtr root = nullptr) {
204 resetDiscovered();
205 std::string return_string;
206
207 // If the argument is not specified, then just use the root.
208 if (root == nullptr) {
209 root = root_;
210 }
211
212 // If the root itself is not found then we can't run DFT.
213 if (root == nullptr) {

215 }
216
217 std::stack<TreeNodePtr> visit{};
218 visit.push(root);
219
220 while (!visit.empty()) {
221 auto &node{visit.top()};
222 node->visit();

224 return_string += " ";

227 // For repeated calls to dft...

233 // Call back function.
234 visit.pop();

236 if (!node->isDiscovered()) {

244 auto const &edges{source->second};
